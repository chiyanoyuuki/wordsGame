      let totalQPAsc = 0;
      let totalQPSkill = 0;
      let totalQPAppend = 0;
      const matsHelper = (className, counterArray, multiplier) => {
        $("." + className + " .paragraph--type--required-materials").each(function(){
          const item = $(this).data("item");
          const qty = parseInt($(this).data("qty")) * multiplier;
          if(counterArray[item]){
            //just add to the existing counter
            counterArray[item]["qty"] = counterArray[item]["qty"] + parseInt(qty);
          }
          else{
            //add to the existing counter and also store the html to output
            counterArray[item] = {};
            counterArray[item]["html"] = $(this)[0].innerHTML;
            counterArray[item]["qty"] = parseInt(qty);
          }
        });
      }
      const sumTotals = (totals) => {
        const toReturn = {};
        for(const total of totals){
          for(const item in total){
            if(toReturn[item]){
              toReturn[item].qty += total[item].qty;
            }
            else{
              toReturn[item] = {};
              toReturn[item].html = total[item].html;
              toReturn[item].qty = total[item].qty;
            }
          }
        }
        return toReturn;
      }
      const printTotalMats = (element, totalArray) => {
        for(key in totalArray){
          var currItemDiv = document.createElement("div");
          currItemDiv.className = "paragraph--type--required-materials";
          currItemDiv.innerHTML = totalArray[key]["html"];
          element.appendChild(currItemDiv);
          $(currItemDiv).find(".field--name-field-number-of-materials").html("x" + totalArray[key]["qty"]);
        }
      }
      $(document).ready(function(){
        //summing total QP
        $(".ascension-cost div").each(function(){
          totalQPAsc += parseInt($(this).attr("content"));
        });
        $("#Skill-materials-table .Skill-cost").each(function(){
          totalQPSkill += parseInt($(this).html().replace(/,/g, ""));
        });
        $("#append-Skill-materials-table .Skill-cost").each(function(){
          totalQPAppend += parseInt($(this).html().replace(/,/g, ""));
        });
        totalQPSkill *= 3;
        totalQPAppend *= 3;
        totalQP = totalQPAsc + totalQPSkill;
        totalTotalTotalQP = totalQP + totalQPAppend;
        matsHelper("ascension-materials", ascensionTotals, 1, totalTotals);
        matsHelper("Skill-materials", skillTotals, 3, totalTotals);
        matsHelper("append-Skill-materials", appendTotals, 3);

        totalTotals = sumTotals([ascensionTotals, skillTotals]);
        totalTotalTotals = sumTotals([totalTotals, appendTotals]); 

        const ascensionEle = document.getElementById("ascension-total");
        printTotalMats(ascensionEle, ascensionTotals);
        createQPDiv(ascensionEle, totalQPAsc);

        const skillEle = document.getElementById("skill-total");
        printTotalMats(skillEle, skillTotals);
        createQPDiv(skillEle, totalQPSkill);

        const totalEle = document.getElementById("total-total");
        printTotalMats(totalEle, totalTotals);
        createQPDiv(totalEle, totalQP);

        const appendEle = document.getElementById("append-total");
        if(appendEle){
          printTotalMats(appendEle, appendTotals);
          createQPDiv(appendEle, totalQPAppend);

          const totalTotalTotalsEle = document.getElementById("total-total-total");
          printTotalMats(totalTotalTotalsEle, totalTotalTotals)
          createQPDiv(totalTotalTotalsEle, totalTotalTotalQP);
        }
        


      });

    function createQPDiv(container, qp){
      //  /sites/grandorder/files/2017-10/Qp.png
      var qpDiv = document.createElement("div");
      qpDiv.className = "total-qp-div clearfix";


      var qpImg = document.createElement("img");
      qpImg.src = "/sites/grandorder/files/2017-10/Qp.png";
      qpDiv.appendChild(qpImg);

      var amtDiv = document.createElement("div");
      amtDiv.className = "qp-amt";
      amtDiv.innerHTML = "x" + qp.toLocaleString();
      qpDiv.appendChild(amtDiv);

      container.appendChild(qpDiv);
    }

<h2 class="main-title">Overview</h2>
<p><span><span><span><span><em><span>“Please come play with the mermaid.”</span></em></span></span></span></span></p>
<p><span><span><span><span><span><span>What happens when you take a lewd nun, a magical girl, and a mermaid, and mix them all together? Sesshouin Kiara (Moon Cancer). As one of the premiere </span></span></span></span></span></span><span><span><span><strong><span><span>AoE Arts Noble Phantasm</span></span></strong></span></span></span><span><span><span><span><span><span> Servants capable of looping, Masters are not going to want to miss out on her… tail.</span></span></span></span></span></span></p>
<p><span><span><span><span><span><span>Kiara has all the makings of a powerhouse. She is a natural Arts looper under the double Castoria setup thanks to her </span></span></span></span></span></span><span><span><span><strong><span><span>Divine Power (Ink)</span></span></strong></span></span></span><span><span><span><span><span><span>’s battery and </span></span></span></span></span></span><span><span><span><strong><span><span>Clam Palace</span></span></strong></span></span></span><span><span><span><span><span><span>’s Arts resist down. She has nice durability too with a Guts skill, an Evasion, and optionally a second stackable Guts on her bond CE. But her true powers lie in her </span></span></span></span></span></span><span><span><span><strong><span><span>Mermaid’s Flesh </span></span></strong></span></span></span><span><span><span><span><span><span>power, which is a unique skill capable of improving any of her other skills as needed, giving her incredible versatility.</span></span></span></span></span></span></p>
<p><span><span><span><span><span><span>On the other hand, Kiara still is a Moon Cancer Servant. While her damage isn’t particularly bad, it will struggle to keep up with Servants who’d otherwise have class advantage. She also is not likely to produce much of anything in the way of a Critical hit, making all her damage focused around her Noble Phantasm. Her natural NP gain is also fairly weak, making her heavily reliant on supports.</span></span></span></span></span></span></p>
<p><span><span><span><span><span><span>Any Master with a Castoria of their own is going to want her, but she’s fairly powerful in her own right. Just be careful… she’s known to take people underwater and they won’t be seen again for decades!</span></span></span></span></span></span></p>
<h2 class="main-title">Strengths</h2>
<div id="servant-strengths">
<h3><span><span><span><span><span><span>Arts Looping</span></span></span></span></span></span></h3>
<p><span><span><span><span><span><span>With a double Castoria setup, Kiara can loop against pretty much any enemy set. While its requirements may differ (see gameplay tip), she has the capacity against all standard 3-3-x nodes without needing to switch in other Servants. </span></span></span></span></span></span></p>
<h3><span><span><span><span><span><span>Strong Durability</span></span></span></span></span></span></h3>
<p><span><span><span><span><span><span>In her natural set, Kiara has both a guts and an evasion, both for one time on relatively short cooldowns. She also has a fairly high HP base, and her Moon Cancer typing means she will very rarely have Supereffective hits levied against her. Her ability to heal herself and remove her debuffs further contributes to this. And if one wanted even more, her bond CE provides a 1-time guts with 100% HP revive to add to this beefiness. </span></span></span></span></span></span></p>
<h3><span><span><span><span><span><span>Versatility</span></span></span></span></span></span></h3>
<p><span><span><span><span><span><span>Mermaid’s Flesh is a unique skill, providing rank ups to her other 2 skills. She will gain 1 stack of rank up each turn, and when she’s ready to use it she selects the skill and it will consume up to 2 stacks of these rank ups. This allows her to choose if she wants a weaker 30% gauge charge or a stronger 50% later, and how much debuffing she wishes to do on her enemies before casting her Noble Phantasm. Good things come to those who wait. </span></span></span></span></span></span></p>
</div>
<h2 class="main-title">Weaknesses</h2>
<div id="servant-weaknesses">
<h3><span><span><span><span><span><span>Mediocre Damage</span></span></span></span></span></span></h3>
<p><span><span><span><span><span><span>Kiara doesn’t have much in terms of damage buffs, with only a 20% NP strength modified. Most of her damage increases rely on her 3rd skill’s debuffs on her enemies, which only lasts the current wave and in 2004-looping strategies will not be available for the toughest 3rd waves against Assassins and Berserkers. Her Moon Cancer classing makes it unlikely she’ll ever see a true usage of class advantage outside of Berserkers. That said, she does have the best damage of any Moon Cancers, so one should always use her against Avengers when they appear. </span></span></span></span></span></span></p>
<h3><span><span><span><span><span><span>Poor Criticals</span></span></span></span></span></span></h3>
<p><span><span><span><span><span><span>Kiara’s Critical capacity is her ability to generate a small pool of stars each turn and a tiny Independent Manifestation C damage buff. All her damage will come from her Noble Phantasm, so Masters should not rely on her cards for any real damage beyond that.</span></span></span></span></span></span></p>
<h3><span><span><span><span><span><span>Low Natural NP Gain</span></span></span></span></span></span></h3>
<p><span><span><span><span><span><span>Without the double Castoria looping strategy, Kiara struggles to get her gauge up more than once. Her base NP gain rate of 0.6% per hit is quite low, and with only a 4-hit Quick and 5-hit Extra she will not generate a ton of gauge on her own. She’ll be reliant on supports for the foreseeable future.</span></span></span></span></span></span></p>
</div>

<h2 class="main-title">Gameplay Tips</h2>
<p><span><span><span><span><span><span>Kiara’s Noble Phantasm, </span></span></span></span></span></span>Nirmāṇarati Heaven's Foam<span><span><span><span><span><span> A, gains power based on how many mental debuff stacks the enemy has, to the effect of 20% per stack. Kiara can use her third skill to inflict stacks of mental debuffs (typically 2 stacks on the third wave after a Mermaid's Flesh buff) and increase damage. However, her third skill is also tied in with her Arts Down, which is used for looping. So one needs to figure out when and how each can be used.</span></span></span></span></span></span></p>
<p><span><span><span><span><span><span>There are three different looping strategies for Kiara based off of class typing with the Fragment of 2004 Mystic Code. All assume double Castoria. Any other looping strategy involves a plugsuit swap for another charger or buffer.</span></span></span></span></span></span></p>
<ul><li><span><span><span><span><span><span><strong>Vs. Casters</strong> - Black Grail capable. Use the Fragment of 2004 CE to buff her NP gain, and give all Castoria buffs to her. Will generate 100% on wave 1, and can use her second skill after wave 3 to fill in the missing amounts.</span></span></span></span></span></span></li>
<li><span><span><span><span><span><span><strong>Vs. Riders/Sabers/Lancers/Archers</strong> - Requires 2nd Skill Append unlocked and maxed. Black Grail Capable. Use the Fragment of 2004 to buff her NP gain and give all Castoria Buff except for one 20% charge/NP gain buff on wave 1. She will generate 80% gauge. You can use the remaining Castoria 20% buff to fill in the remaining gauge, and on turn 3 use her second skill to fill in the remaining amounts.</span></span></span></span></span></span></li>
<li><span><span><span><span><span><span><strong>Vs. Assassins / Berserkers</strong> - Kiara will have to accept a 3rd wave damage reduction and use her 3rd skill on wave 1, in addition to using a NP gain CE such as Sign of the Smiling Face and all Castoria buffs. Facing non-Casters without the append skill is similar. Most would want to use a plugsuit swap and Black Grail in these situations instead. </span></span></span></span></span></span></li>
</ul>

<h2 class="main-title">Profile</h2>
<div class="paragraph paragraph--type--servant-profile-entry paragraph--view-mode--default">
<h3 class="sub-title" style="margin-top:8px">Character Info</h3>
<p>A beautiful, mysterious woman who claims to be the Pseudo-Servant of the legendary nun Yaobikuni, who was said to have gained immortality after consuming the flesh of a mermaid and then went on a pilgrimage across Japan... And nobody knows anything about her beyond that.</p>
<p>"Hehehe...being in a swimsuit Spirit Origin is certainly amusing."</p>
<p>A dizzyingly dazzling guise of Sessyoin Kiara. Step into her realm, leaving behind your worries, and her beauty will make you lose all sense of time.</p>
</div>
<div class="paragraph paragraph--type--servant-profile-entry paragraph--view-mode--default">
<h3 class="sub-title" style="margin-top:8px">Profile 1</h3>
<p>Height/Weight: 166cm, 53kg<br />
Origin: Mermaid legend, Supplement to The Hundred Demons from the Present and the Past, from "Abyssal Cyber Paradise, SE.RA.PH"<br />
Region: Japan, somewhere under the sea<br />
Alignment: Lawful-Evil<br />
Gender: Female<br />
This time around, she has other transformations in store.<br />
"It's summer. It would be boring if I only had a swimsuit. There should be a few surprises to enjoy, right?"<br />
She's having quite a lot of fun.</p>
</div>
<div class="paragraph paragraph--type--servant-profile-entry paragraph--view-mode--default">
<h3 class="sub-title" style="margin-top:8px">Profile 2</h3>
<p>Even when she attempted to become Yaobikuni, Sessyoin Kiara did not change.</p>
<p>"There is only one of me in this world. All insects, good and bad, are equal. Whatever I find, I shall feel affection for it, heal it, and love it..."</p>
<p>But this love she speaks of is naturally something that pleases Kiara first and foremost. Even should such love seem to be salvation in the eyes of its object, an outsider will see that it is no more than the hell of being manipulated. A demonic being who uses her power as a "savior" solely for her own pleasure... Well, that's Sessyoin Kiara for you.</p>
<p>...At least that's who she is. But maybe because so many things happened this time, she is a little more thoughtful regarding how others perceive her, if only slightly. Perhaps her attempt to act as the nun Yaobikuni has taught her some measure of restraint.</p>
</div>
<div class="paragraph paragraph--type--servant-profile-entry paragraph--view-mode--default">
<h3 class="sub-title" style="margin-top:8px">Profile 3</h3>
<p>- First Ascension<br />
As this is her first swimsuit and summer event, her innocence and femininity are somewhat more pronounced than in the normal Kiara. She also feels more carefree, and she's secretly elated about taking a summer vacation, as she never had a chance to experience one while she was a child. This has caused a terrifying bug where Master only perceives her as an elegant (and sometimes girlishly cute) lady they met at a summer resort.</p>
<p>- Second Ascension<br />
A lily version of herself who used a makeup compact to transform into a magical girl (or so she claims). In this form, she is more sadistic and merciless than the normal Kiara, if you can believe it. She is a terrifying queenly enforcer who has not a bit of mercy about pointing out the flaws of any impudent or masochistic little pigs (regardless of gender) who had it coming in her bid to send them all to ruin. How did she end up like this...well, this lily has quite a dark side.</p>
<p>- Third Ascension<br />
It's the usual Kiara...but she is in a long-anticipated mermaid cosplay and her "lawful" side is more pronounced, so she's actually a bit less wicked, and instead her prim and proper side is front and center. Considering she's more of a compassionate bodhisattva when merged with a monster, how might the fairy tale writer chastise her in this form?</p>
</div>
<div class="paragraph paragraph--type--servant-profile-entry paragraph--view-mode--default">
<h3 class="sub-title" style="margin-top:8px">Profile 4</h3>
<p>Otherworldly Creation: EX<br />
As she harbors a monstrosity within herself, it is possible for her to create a phantom world atop our current reality.</p>
<p>Logos Eater: D<br />
A special quality as the deity of pleasure. The ideology of toying with intelligent life. Its rank is lower than with the normal Kiara.</p>
<p>Divine Power (Ink): B<br />
Perhaps she was jealous, or maybe she was struck by inspiration... Originally the skill was the normal Divine Power, but after seeing Katsushika Hokusai (Saber)'s chic and stylish Divine Power, she couldn't help but mimic it.</p>
<p>Mermaid's Flesh: EX<br />
Legend has it that anyone who eats mermaid flesh becomes immortal. In Japan, mermaids are believed to lack intelligence and be incapable of speaking. They are seen simply as fish in the shape of human beings. That is to say, they have no soul. And thus eating a fish with no soul would not count as taking a life, and so the girl's vow remains unbroken. ...Even if she eats an entire school of mermaids.</p>
<p>As a side note, a certain fairy tale writer once wrote that mermaids are as intelligent as humans, but lack a soul.</p>
</div>

  function getRank(array, stat){
    var rank = -1;
    var count = 0;
    for(var i = 0; i < array.length; i++){
      if(array[i] == stat){
        rank = i+1;
        return rank;
      }
    }
  }
  function getInfo(array, field){
    var uniques = [];
    var highest = -1;
    for(var i = 0; i < array.length; i++){
      var currVal = array[i][field];
      var currValInt = parseInt(currVal);
      var currClass = array[i].tdoll_class;
      if(currValInt > highest){
        highest = currValInt;
      }
      uniques.push(currVal);
    }
    return [uniques, highest];
  }

function sortJSONByField(json, field){
  return json.concat().sort(function(a, b){
    return parseInt(b[field]) - parseInt(a[field]);
  });
}

      var modal = document.getElementById('gp-tier-modal');
